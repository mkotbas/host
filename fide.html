function loadReportFromEmail() { if (!selectedStore) { alert('Lütfen e-posta içeriğini yüklemeden önce denetlenecek bayiyi seçin!'); return; } const emailContent = document.getElementById('email-report-area').value; if (!emailContent) { alert('Lütfen e-posta içeriğini yapıştırın.'); return; } // E-posta içeriğini satırlara ayır ve tüm metin içeriğini temizle const cleanContent = emailContent.replace(/<[^>]*>/g, '').replace(/ /g, '').trim(); const lines = cleanContent.split('\n'); const questions = {}; let currentId = null; // Metni FiDe başlıklarına göre böl let buffer = ''; lines.forEach(line => { const trimmedLine = line.trim(); const match = trimmedLine.match(/FiDe (\d+)\./); if (match) { // Yeni bir FiDe başlığı bulundu if (currentId !== null && buffer.trim() !== '') { // Bir önceki sorunun içeriğini işle questions[currentId] = buffer.trim(); } currentId = match[1]; buffer = trimmedLine + '\n'; } else if (currentId !== null) { // Aynı sorunun alt maddelerini biriktir buffer += trimmedLine + '\n'; } }); // Son sorunun içeriğini işle if (currentId !== null && buffer.trim() !== '') { questions[currentId] = buffer.trim(); } // Formu temizle ve yeni verilerle doldur resetForm(); const formQuestions = document.querySelectorAll('.fide-item'); formQuestions.forEach(qElement => { const qId = qElement.id.replace('fide-item-', ''); const questionData = questions[qId]; if (questionData) { // E-postada ilgili soru varsa, içeriğini işle const items = questionData.split('\n') .map(line => line.replace(/^(\*|\-|•)\s*/, '').trim()) // 'Tamamlandı' ifadesi içeren satırları filtrele .filter(line => !line.includes('Tamamlandı')) .filter(line => line !== '' && !line.includes('SİRKÜYÜ İNCELEMEK İÇİN TIKLAYINIZ.') && !line.includes('ÖRNEK FOTOĞRAF İÇİN TIKLAYINIZ.') && !line.includes('Tüm ürünlerde enerji etiketi var mı?') && !line.includes('Aşağıdaki ürünlerin enerji etiketlerinin tamamlanmasını rica ederim.') && !line.includes('Ürünler teşhir ürün dizilim kılavuzuna göre sergilenmiş mi?') && !line.includes('Aşağıdaki ürünlerin teşhir ürün dizilim kılavuzuna (5 nolu sirkü) göre düzenlenmesini rica ederim.') && !line.includes('@tevfik_2979@hotmail.com')) .filter(line => line.toLowerCase().includes('eksik') || !fideQuestions.find(q => q.id == qId).title.toLowerCase().includes(line.toLowerCase())); const questionInfo = fideQuestions.find(q => String(q.id) === qId); const containerId = (questionInfo && questionInfo.type === 'product_list') ? `fide${qId}_pleksi` : `fide${qId}`; const staticItems = questionInfo.staticItems || []; const normalizedStaticItems = staticItems.map(s => s.trim().replace(/<[^>]*>/g, '').replace(/ /g, '').toLowerCase()); items.forEach(item => { const questionTitle = fideQuestions.find(q => q.id == qId).title; if (item.includes(questionTitle) || item.includes('rica ederim')) { return; } // Eğer madde statik maddeler arasında yoksa ekle if (!normalizedStaticItems.includes(item.toLowerCase())) { addDynamicInput(containerId, item, false); } }); } else { // E-postada ilgili soru yoksa, formdaki öğeyi kaldır (removed) const removeButton = qElement.querySelector('.remove-btn'); if (removeButton && !qElement.classList.contains('question-removed')) { toggleQuestionRemoved(removeButton, qId); } } }); saveFormState(); alert('E-posta içeriği başarıyla işlenerek forma aktarıldı. Tamamlandı olarak işaretlenmesi gerekenleri manuel olarak güncelleyebilirsiniz.'); }
